1) display ename,job,sal,deptno,dname whose location is "CHICAGO"
select e.ename,e.job,e.sal,d.deptno,d.dname,d.loc from emp e,dept d where e.deptno = d.deptno and d.loc="CHICAGO"
2) display number of employees whose department is "ACCOUNTING"
select count(1) emp_cnt from emp e where deptno in (select deptno in dept whare dname="ACCOUNTING")
3) display ename,job,sal,dname whose dname 5th character should be "S"
select e.ename,e.job,e.sal,d.dname from emp e,dept d where e.deptno = d.deptno and d.dname like "____S%"
