1) Display employee details whose employee name contains Letter "S" 
   and also display employee details whose location is "NEW YORK"
   
   SELECT * FROM EMP WHERE ENAME LIKE '%S%'
   UNION ALL
   SELECT * FROM EMP WHERE DEPTNO IN(SELECT DEPTNO FROM DEPT WHERE LOC ='NEW YORK')
   
2) Display employee details whole employee name starts with "A" or fourth letter shoud be "M" 
   and also display employee details whose location is "NEW YORK" 
   and hiredate between "01-JAN-1981" AND "31-DEC-1981"
   
   SELECT * FROM EMP WHERE (ENAME LIKE 'A%' OR ENAME LIKE '___M%')
   UNION ALL
   SELECT * FROM EMP WHERE DEPTNO IN(SELECT DEPTNO FROM DEPT WHERE LOC='NEW YORK')
   AND HIREDATE between TO_DATE('01-JAN-1981','DD-MON-YYYY') 
   AND TO_DATE('31-DEC-1981','DD-MON-YYYY')
   
3) Display employee name and their manager name whose employee name ending with letter "S"
   and manager name either 4th letter should be "T" 
   or manager name 4th letter should be letter "G"
   
   SELECT 
   E.ENAME EMPLOYEE,
   M.ENAME MANAGER
   FROM
   EMP E,
   EMP M
   WHERE
   E.MGR = M.EMPNO
   AND E.ENAME LIKE '%S'
   AND (M.ENAME LIKE '___T%' OR M.ENAME LIKE '___G%')
   
4) display employee details whose salary between 800 AND 2000 
   and also Display employee details department no in 10 and 30
   
   select * from emp where sal between 800 AND 2000
   UNION ALL
   SELECT * FROM EMP WHERE DEPTNO IN(10,30)
